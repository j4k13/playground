function submit_view()
{
var raw = $('#add_view_form').serializeArray();

var result = {};
for(var index = 0; index < raw.length; index++)
{
result[ raw[index].name ] = raw[index].value;
}

// Validate input
if(!result.name || /[^A-Za-z\d ]/.test(result.name))
{
$('#views_guidance').html('Please use numbers and English letters to describe the location.');
return false;
}

hide_add_view();
}
